For example: &nbsp; int u = 2; &nbsp; int v = 3; &nbsp; int w = 4; &nbsp; int x = 1; &nbsp; sort2(u, v); &nbsp; //u is still 2, v is still 3// &nbsp; sort2(w, x); &nbsp; //w is now 1, x is now 4//
